(defvar elpaca-installer-version 0.12)

(defvar elpaca-order '(elpaca :repo "https://github.com/progfolio/elpaca.git"
                              :ref nil :depth 1 :inherit ignore
                              :files (:defaults "elpaca-test.el" (:exclude "extensions"))
                              :build (:not elpaca-activate)))
(let* ((repo  (expand-file-name "elpaca/" elpaca-sources-directory))
       (build (expand-file-name "elpaca/" elpaca-builds-directory))
       (order (cdr elpaca-order))
       (default-directory repo))
  (add-to-list 'load-path (if (file-exists-p build) build repo))
  (unless (file-exists-p repo)
    (make-directory repo t)
    (when (<= emacs-major-version 28) (require 'subr-x))
    (condition-case-unless-debug err
        (if-let* ((buffer (pop-to-buffer-same-window "*elpaca-bootstrap*"))
                  ((zerop (apply #'call-process `("git" nil ,buffer t "clone"
                                                  ,@(when-let* ((depth (plist-get order :depth)))
                                                      (list (format "--depth=%d" depth) "--no-single-branch"))
                                                  ,(plist-get order :repo) ,repo))))
                  ((zerop (call-process "git" nil buffer t "checkout"
                                        (or (plist-get order :ref) "--"))))
                  (emacs (concat invocation-directory invocation-name))
                  ((zerop (call-process emacs nil buffer nil "-Q" "-L" "." "--batch"
                                        "--eval" "(byte-recompile-directory \".\" 0 'force)")))
                  ((require 'elpaca))
                  ((elpaca-generate-autoloads "elpaca" repo)))
            (progn (message "%s" (buffer-string)) (kill-buffer buffer))
          (error "%s" (with-current-buffer buffer (buffer-string))))
      ((error) (warn "%s" err) (delete-directory repo 'recursive))))
  (unless (require 'elpaca-autoloads nil t)
    (require 'elpaca)
    (elpaca-generate-autoloads "elpaca" repo)
    (let ((load-source-file-function nil)) (load "./elpaca-autoloads"))))
(add-hook 'after-init-hook #'elpaca-process-queues)

(defun my/tab-line-keep-p (buffer)
  "Keep file buffers + shells; hide most *special* buffers."
  (let ((name (buffer-name buffer)))
    (and (not (string-prefix-p " " name))
         (or (not (string-prefix-p "*" name))
             (with-current-buffer buffer
               (derived-mode-p 'ghostel-mode 'eshell-mode 'shell-mode
                               'term-mode 'comint-mode))))))

(defun my/tab-line-tabs ()
  "Window buffer tabs without special *buffers* (keep current).\nOrder is stable across focus changes (fixed-window-buffers)."
  (let ((cur (current-buffer)))
    (seq-filter (lambda (b) (or (eq b cur) (my/tab-line-keep-p b)))
                (tab-line-tabs-fixed-window-buffers))))

(defun my/agent-notify (message &optional buffer silent)
  "Record agent MESSAGE for BUFFER (default current). Unread, newest first.
If SILENT, skip alert/echo (caller shows notify, e.g. ghostel-default-notify)."
  (interactive "sNotify: ")
  (let* ((buf (or buffer (current-buffer)))
         (name (buffer-name buf)))
    (setq my/agent-notifs
          (cons (list buf (float-time) message t)
                (seq-remove (lambda (n) (eq (car n) buf)) my/agent-notifs)))
    (my/agent-refresh-ui)
    (unless silent
      (alert message :title (format "Emacs · %s" name))
      (message "● notify [%s]: %s" name message))))

(defun my/agent-notif-mode-line ()
  "Mode-line segment: unread agent notif count."
  (let ((n (seq-count (lambda (x) (nth 3 x)) (my/agent-notifs-sorted))))
    (when (> n 0)
      (propertize (format " ●%d" n) 'face 'warning))))
(add-to-list 'mode-line-misc-info '(:eval (my/agent-notif-mode-line)) t)

(defun my/agent-notifs-sorted ()
  "Live notifs: unread first, then newest."
  (sort (seq-filter (lambda (n) (buffer-live-p (car n))) my/agent-notifs)
        (lambda (a b)
          (let ((ua (nth 3 a)) (ub (nth 3 b)))
            (if (eq ua ub)
                (> (nth 1 a) (nth 1 b))
              (and ua (not ub)))))))

(defun my/agent-mark-buffers-read (buffers)
  "Mark agent notifs for BUFFERS as read; refresh UI if anything changed."
  (let ((changed nil))
    (dolist (n my/agent-notifs)
      (when (and (nth 3 n) (memq (car n) buffers))
        (setf (nth 3 n) nil)
        (setq changed t)))
    (when changed (my/agent-refresh-ui))))

(defun my/agent-mark-buffer-read (&optional buffer)
  "Mark agent notif for BUFFER (default current) as read."
  (my/agent-mark-buffers-read (list (or buffer (current-buffer)))))

(defun my/agent-on-show (&optional _frame)
  "Clear unread when selecting / showing a buffer."
  (my/agent-mark-buffer-read (window-buffer (selected-window))))

(defun my/agent-on-tab-select (_from to)
  "Clear unread for all buffers visible in selected workspace tab."
  (when-let ((bufs (my/tab-buffers to)))
    (my/agent-mark-buffers-read bufs)))

(add-hook 'window-selection-change-functions #'my/agent-on-show)
(add-hook 'window-buffer-change-functions #'my/agent-on-show)
(add-hook 'tab-bar-tab-post-select-functions #'my/agent-on-tab-select)

(defun my/agent-jump-unread ()
  "Jump to newest unread agent notification."
  (interactive)
  (if-let ((n (seq-find (lambda (x) (nth 3 x)) (my/agent-notifs-sorted))))
      (progn
        (my/agent-mark-buffer-read (car n))
        (pop-to-buffer (car n)))
    (message "no unread")))

(defun my/agent-list-notifs ()
  "Pick a notification (unread sorted first) and jump."
  (interactive)
  (let* ((items (my/agent-notifs-sorted))
         (cands (mapcar (lambda (n)
                          (cons (format "%s%s  %s"
                                        (if (nth 3 n) "● " "  ")
                                        (buffer-name (car n))
                                        (nth 2 n))
                                n))
                        items)))
    (if (null cands)
        (message "no notifications")
      (let* ((pick (completing-read "notif: " cands nil t))
             (n (cdr (assoc pick cands))))
        (my/agent-mark-buffer-read (car n))
        (pop-to-buffer (car n))))))

(defun my/tab-line-name (buffer &optional buffers)
  "tab-line label: N name; unread → ● N name. Index is per-window."
  (let* ((tabs (or buffers (funcall tab-line-tabs-function)))
         (idx (seq-position tabs buffer))
         (num (if idx (format "%d " (1+ idx)) ""))
         (name (buffer-name buffer))
         (unread (seq-find (lambda (n) (and (eq (car n) buffer) (nth 3 n)))
                           my/agent-notifs)))
    (if unread
        (concat (propertize "● " 'face 'warning) num name)
      (concat num name))))
(setq tab-line-tab-name-function #'my/tab-line-name)

(defun my/tab-buffers (tab)
  "Live buffers shown in TAB's window layout.
Inactive tabs store names in `ws'; normalize to buffer objects."
  (delq nil
        (mapcar (lambda (b)
                  (cond ((bufferp b) (and (buffer-live-p b) b))
                        ((stringp b) (get-buffer b))))
                (if (eq (car tab) 'current-tab)
                    (mapcar #'window-buffer (window-list-1 nil 'nomini t))
                  (when-let ((ws (alist-get 'ws tab)))
                    (window-state-buffers ws))))))

(defun my/tab-has-unread-p (tab)
  "Non-nil if TAB shows a buffer with unread agent notif."
  (when-let ((bufs (my/tab-buffers tab)))
    (seq-some (lambda (n) (and (nth 3 n) (memq (car n) bufs)))
              my/agent-notifs)))

(defun my/tab-bar-sort-unread-first ()
  "Stable-sort workspace tabs: unread notif tabs first."
  (let* ((tabs (funcall tab-bar-tabs-function))
         (sorted (sort (copy-sequence tabs)
                       (lambda (a b)
                         (and (my/tab-has-unread-p a)
                              (not (my/tab-has-unread-p b)))))))
    (unless (equal tabs sorted)
      (tab-bar-tabs-set sorted))))

(defun my/tab-bar-name-format (tab i)
  "tab-bar label; ● if any buffer in TAB has unread agent notif."
  (let ((base (tab-bar-tab-name-format-default tab i)))
    (if (my/tab-has-unread-p tab)
        (concat (propertize "● " 'face 'warning) base)
      base)))
(setq tab-bar-tab-name-format-function #'my/tab-bar-name-format)
